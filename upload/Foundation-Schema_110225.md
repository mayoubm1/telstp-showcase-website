  private agents: Record<string, any>;

  constructor(agentsConfig: any) {
    this.agents = agentsConfig;
  }

  async refreshContext(agentName: string, newPayload: any) {
    const storedContext = await AsyncStorage.getItem(`${agentName}_context`);
    const fullPayload = { ...JSON.parse(storedContext || '{}'), ...newPayload };
    // Call Mistral API (or proxy) for processing
    const response = await fetch('https://api.mistral.ai/v1/chat/completions', { // Use your keys
      method: 'POST',
      headers: { 'Authorization': `Bearer ${this.agents[agentName].apiKey}` },
      body: JSON.stringify({ model: 'mistral-large', messages: [{ role: 'user', content: JSON.stringify(fullPayload) }] })
    });
    const result = await response.json();
    // Save summary
    await AsyncStorage.setItem(`${agentName}_context`, JSON.stringify({ summary: result.choices[0].message.content }));
    return result;
  }
}
